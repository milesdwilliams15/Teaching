\begin{frame}{What is it?}
\protect\hypertarget{what-is-it-1}{}
\begin{enumerate}
\item
  Countries should accept the borders of other countries and not use
  force in an attempt to change them.
\item
  Countries should not interfere with events inside other countries.
\item
  Governments should have a free hand to do as they please in their own
  borders.
\end{enumerate}

\begin{frame}{What is it?}
\protect\hypertarget{what-is-it-3}{}
\begin{enumerate}
\item
  The ``citizens'' of this society are \textbf{\emph{countries}}.
\item
  Governments and leaders of these countries are free to act as they
  wish within their own borders.
\item
  The members of this society respect and accept this freedom of action
  on the part of others.
\end{enumerate}
\end{frame}
